Code pour gnrer (5000+lignes) lignes alatoires

In[16]:
import pandas as pd
import numpy as np
from faker import Faker
fake = Faker() # Gnrateur de donnes alatoires
# Nombre de lignes  gnrer
n = 5000
# Gnrer des donnes alatoires
data = {
'ID': np.arange(1, n+1), # ID de 1  5000
'Nom': [fake.name() for _ in range(n)], # Noms alatoires
'ge': np.random.randint(18, 60, size=n), # ge entre 18 et 60 ans
'Ville': [fake.city() for _ in range(n)], # Villes alatoires
'Email': [fake.email() for _ in range(n)], # Emails alatoires
'Date Inscription': [fake.date_this_decade() for _ in range(n)] # Dates d'inscription
}
# Convertir en DataFrame Pandas
df = pd.DataFrame(data)
# Enregistrer dans un fichier Excel
df.to_excel('grande_base_de_donnees.xlsx', index=False, engine='openpyxl')
print("Fichier Excel de 5000 entres gnr avec succs !")

Fichier Excel de 5000 entres gnr avec succs !

import matplotlib.pyplot as plt
import numpy as np
from mpl_toolkits.mplot3d import Axes3D
# Gnrer une grille pour X et Y
x = np.linspace(-5, 5, 100)
y = np.linspace(-5, 5, 100)
X, Y = np.meshgrid(x, y)
# Dfinir la fonction Z (par exemple, une surface sinusodale)
In[23]:
Z = np.sin(np.sqrt(X**2 + Y**2))
# Crer la figure et le subplot 3D
fig = plt.figure(figsize=(8, 6))
ax = fig.add_subplot(111, projection='3d')
# Tracer les contours 3D
ax.contour3D(X, Y, Z, 50, cmap='viridis')
# Ajouter des labels et titre
ax.set_xlabel("X")
ax.set_ylabel("Y")
ax.set_zlabel("Z")
ax.set_title("Contours 3D de la fonction Z = sin(sqrt(X2 + Y2))")
plt.show()
